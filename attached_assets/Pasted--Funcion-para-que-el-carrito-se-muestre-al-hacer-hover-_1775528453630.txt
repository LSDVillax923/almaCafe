//Lista de compras
let listaDeCompras = [];


//Funcion para agregar productos al carrito
function agregarAlCarrito(nombre, precio, imagen){
  const producto = listaDeCompras.find(p => p.nombre === nombre);

  //Si el producto ya existe, se aumenta la cantidad
  if(producto){
    producto.cantidad++;
  } else {
    listaDeCompras.push({
      nombre,
      precio,
      imagen,
      cantidad: 1
    });
  }
//Se renderiza el carrito
  renderCarrito();
}

//Funcion para renderizar el carrito
function renderCarrito(){
  const lista = document.getElementById("lista-carrito");
  lista.innerHTML = "";
//Se recorre la lista de compras y se agrega cada producto al carrito
  listaDeCompras.forEach((item, index) => {
    lista.innerHTML += `
      <tr>
        <td><img src="${item.imagen}" width="40"></td>
        <td>${item.nombre}</td>
        <td>$${item.precio}</td>
        <td>
          <button onclick="cambiarCantidad(${index}, -1)">➖</button>
          ${item.cantidad}
          <button onclick="cambiarCantidad(${index}, 1)">➕</button>
        </td>
      </tr>
    `;
  });
}


//Funcion para vaciar el carrito
function vaciarCarrito(){
  listaDeCompras = [];
  document.getElementById("lista-carrito").innerHTML = " ";
}

//Funcion para agregar un nuevo articulo
const form = document.getElementById("form-articulo");
const contenedor = document.querySelector(".articulos");

form.addEventListener("submit", function(e){
  e.preventDefault();

  const nombre = document.getElementById("nombre").value;
  const precio = Number(document.getElementById("precio").value);
  const imagen = document.getElementById("imagen").value;
  const attr1 = document.getElementById("attr1").value;
  const attr2 = document.getElementById("attr2").value;
  const attr3 = document.getElementById("attr3").value;
  //Validacion de precio
  if(precio < 1000){
    alert("El precio debe ser mínimo 1000");
    return;
  }
  //Validacion de imagen
  const articulo = document.createElement("article");
  articulo.classList.add("articulo");

  //Se agrega el articulo al contenedor
  articulo.innerHTML = `
    <img src="${imagen}" height="150px" alt="${nombre}" />

    <div class="card-info">
      <h3>${nombre}</h3>

      <ul>
        <li>${attr1}</li>
        <li>${attr2}</li>
        <li>${attr3}</li>
      </ul>

      <p class="precio">$ ${precio}</p>

      <button>
        Agregar al carrito
      </button>
    </div>
  `;

  //Se agrega el evento al boton de agregar al carrito
  articulo.querySelector("button").addEventListener("click", ()=>{
    agregarAlCarrito(nombre, precio, imagen);
  });

  //Se agrega el articulo al contenedor
  contenedor.appendChild(articulo);
//Se resetea el formulario
  form.reset();
});

//Funcion para cambiar la cantidad de un producto
function cambiarCantidad(index, cambio){
  listaDeCompras[index].cantidad += cambio;
//Si la cantidad es menor o igual a 0, se elimina el producto
  if(listaDeCompras[index].cantidad <= 0){
    listaDeCompras.splice(index,1);
  }
//Se renderiza el carrito
  renderCarrito();
}
